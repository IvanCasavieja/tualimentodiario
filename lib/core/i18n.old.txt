final stringsProvider = Provider<Strings>((ref) {
  final lang = ref.watch(languageProvider);
  switch (lang) {
    case AppLang.en:
      return _en;
    case AppLang.pt:
      return _pt;
    case AppLang.it:
      return _it;
    case AppLang.es:
      return _es;
  }
});
